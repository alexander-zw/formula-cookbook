\noindent
\begin{tabular}{c|cccc}
    $X$ & Range & $\Pb(X=k)$ & $\E(X)$ & $\var(X)$ \\
    \hline
    Uniform($a,b$) & $a..b$ & $\ds\frac1{b-a+1}$ & $\ds\frac{a+b}2$ & $\ds\frac{(b-a+1)^2-1}{12}$ \\

    \end{cases}$ & $p$ & $p(1-p)$ \\
